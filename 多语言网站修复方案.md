  private async validatePageContent(result: SyncValidationResult) {

        const enStat = fs.statSync(enPath);

        if (zhStat.mtime > enStat.mtime) {
          result.outdated.push(`pages/${page}/en.json`);
          result.synced = false;
        }
      }
    }
  }

  private async validateNewsContent(result: SyncValidationResult) {
    const newsPath = path.join(process.cwd(), 'src/content/news');

    if (!fs.existsSync(newsPath)) return;

    const newsItems = this.getDirectories(newsPath);

    for (const newsId of newsItems) {
      const zhPath = path.join(newsPath, newsId, 'zh.md');
      const enPath = path.join(newsPath, newsId, 'en.md');

      if (!fs.existsSync(zhPath)) {
        result.missingZh.push(`news/${newsId}/zh.md`);
        result.synced = false;

      if (!fs.existsSync(enPath)) {
        result.missingEn.push(`news/${newsId}/en.md`);
        result.synced = false;
      }
    }

  private async validateTranslationFiles(result: SyncValidationResult) {
    const translationPath = path.join(process.cwd(), 'src/contexts/LanguageContext.tsx');

    if (fs.existsSync(translationPath)) {
      const content = fs.readFileSync(translationPath, 'utf8');

      // æ£€æŸ¥ç¿»è¯‘å®Œæ•´æ€§
      const zhMatches = content.match(/zh:\s*{([^}]+)}/g);
      const enMatches = content.match(/en:\s*{([^}]+)}/g);

        const zhKeys = this.extractKeys(zhMatches.join(''));
        const enKeys = this.extractKeys(enMatches.join(''));

        const missingInEn = zhKeys.filter(key => !enKeys.includes(key));
        const missingInZh = enKeys.filter(key => !zhKeys.includes(key));

        if (missingInEn.length > 0) {
          result.missingEn.push(...missingInEn.map(key => `translations.${key}`));
          result.synced = false;

        if (missingInZh.length > 0) {

          result.synced = false;
        }

  private extractKeys(content: string): string[] {

      keys.push(match[1]);
    }

    return keys;

    return fs.readdirSync(dirPath).filter(file => {
      const fullPath = path.join(dirPath, file);
      return fs.statSync(fullPath).isDirectory();
    });
  }
